#### 1 结构体做函数参数
```cpp
// 创建学生数据类型：学生
struct Student {
    //成员列表，(姓名，年龄，分数)
    string name;
    int age;
    int score;
};
```
- **值传递**
```cpp
// 值传递函数
void test01(Student stu) {
    stu.score = 100;
    cout << "姓名：" << stu.name << " 年龄：" << stu.age << " 分数：" << stu.score << endl;
}

int main() {    
    // 创建结构体变量
    Student A = {"张三", 18, 99};
    cout << "姓名：" << A.name << " 年龄：" << A.age << " 分数：" << A.score << endl;
    test01(A);
    cout << "姓名：" << A.name << " 年龄：" << A.age << " 分数：" << A.score << endl;
}
```
stdout:
```
姓名：张三 年龄：18 分数：99
姓名：张三 年龄：18 分数：100
姓名：张三 年龄：18 分数：99
```

- **指针传递**
```cpp
// 指针传递函数
void test02(Student* stuPtr) {
    stuPtr->score = 100;
    cout << "姓名：" << stuPtr->name << " 年龄：" << stuPtr->age << " 分数：" <<
        stuPtr->score << endl;
}

int main() {    
    // 创建结构体变量
    Student A = {"张三", 18, 99};
    cout << "姓名：" << A.name << " 年龄：" << A.age << " 分数：" << A.score << endl;
    test02(&A);
    cout << "姓名：" << A.name << " 年龄：" << A.age << " 分数：" << A.score << endl;
}
```
stdout:
```
姓名：张三 年龄：18 分数：99
姓名：张三 年龄：18 分数：100
姓名：张三 年龄：18 分数：100
```
